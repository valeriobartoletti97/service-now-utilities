api.controller=function() {
  /* widget controller */
  var c = this;
  c.modal = false;
  c.name = "";
  c.productList = [
	{
		label : "Smartphone",
		value : "smartphone",
	},
	{
		label : "Computer",
		value : "computer",
	},
	{
		label : "Tv",
		value : "tv",
	}
  ]
  c.productType="";
  c.price = "";

  c.insertProduct = function() {
        c.server.get({
			parameter1: c.name,
			parameter2: c.productType,
			parameter3: c.price,
            action: "insertProduct"
        }).then(function(response) {
            //Process your response
            c.data.response = response.data.insertResponse;
            if (c.data.response){
				alert("Prodotto aggiunto");
				location.reload();
			} else {
				alert("Si Ã¨ verificato un problema: prodotto non aggiunto");
			}
			c.modal = false;
        });
    };
};

var SpInsertProduct = Class.create();
SpInsertProduct.prototype = Object.extendsObject(AbstractAjaxProcessor, {
	insertProduct: function(name,type,price){
		var gr = new GlideRecord("u_prodotto");
		gr.initialize();
		gr.u_name = name;
		gr.u_type = type;
		gr.u_prezzo = price;
		var insertOk = gr.insert();
		if (insertOk){
			return true;
		} else {
			return false;
		}
	},
    type: 'SpInsertProduct'
});
